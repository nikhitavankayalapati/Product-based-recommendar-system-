        "def similar_users(user_index, interactions_matrix):\n",
        "    similarity = []\n",
        "    for user in range(0, interactions_matrix.shape[0]): #  .shape[0] gives number of rows\n",
        "\n",
        "        #finding cosine similarity between the user_id and each user\n",
        "        sim = cosine_similarity([interactions_matrix.loc[user_index]], [interactions_matrix.loc[user]])\n",
        "\n",
        "        #Appending the user and the corresponding similarity score with user_id as a tuple\n",
        "        similarity.append((user,sim))\n",
        "\n",
        "    similarity.sort(key=lambda x: x[1], reverse=True)\n",
        "    most_similar_users = [tup[0] for tup in similarity] #Extract the user from each tuple in the sorted list\n",

        "\n",

        "    most_similar_users.remove(user_index)\n",
        "    similarity_score.remove(similarity_score[0])\n",

      "source": [
        "def recommendations(user_id, num_recommendations, ratings_matrix):\n",
        "    # Step 1: Find similar users (assumes similar_users returns a list of user IDs)\n",
        "    similar_user_ids = similar_users(user_id, ratings_matrix)[0][:10]  # take top 10 similar users\n",
        "\n",
        "    # Step 2: Get ratings from those similar users\n",
        "    similar_users_ratings = ratings_matrix.loc[similar_user_ids]\n",
        "\n",
        "    # Step 3: Calculate the average rating per product from similar users\n",
        "    mean_product_ratings = similar_users_ratings.mean(axis=0)\n",
        "\n",
        "    # Step 4: Remove products already rated by the target user\n",
        "    user_rated_items = ratings_matrix.loc[user_id][ratings_matrix.loc[user_id].notna()].index\n",
        "    mean_product_ratings = mean_product_ratings.drop(user_rated_items, errors='ignore')\n",
        "\n",
        "    # Step 5: Recommend the top-N highest-rated products\n",
        "    recommended_items = mean_product_ratings.sort_values(ascending=False).head(num_recommendations)\n",
        "\n",
        "    return recommended_items\n"
